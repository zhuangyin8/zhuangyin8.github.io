function DataType(tgt, type) {
  const dataType = Object.prototype.toString
    .call(tgt)
    .replace(/\[object (\w+)\]/, '$1')
    .toLowerCase();
  return type ? dataType === type : dataType;
}
DataType('young'); // "string"
DataType(20190214); // "number"
DataType(true); // "boolean"
DataType([], 'array'); // true
DataType({}, 'array'); // false
function DataType(tgt, type) {
  const dataType = Object.prototype.toString

}

混淆数组 数组乱序

// ❌
const arr = [0, 1, 2, 3, 4, 5].slice().sort(() => Math.random() - 0.5);
// arr => [3, 4, 0, 5, 1, 2]
// 洗牌算法
function shuffle(arr) {
  let i = arr.length;
  while (i) {
    let j = Math.floor(Math.random() * i--);
    [arr[j], arr[i]] = [arr[i], arr[j]];
  }
  return arr;
}

console.log(shuffle([1, 2, 3, 4, 5, 6, 7, 8, 9]));
答案

分析:

⚠️ 这道题检验了数组的 sort() 方法,因为是乱序,所以还需要用到 Math.random() 产生随机数,打乱排序规律,但要注意直接利用 sort 进行排序, 乱序结果概率不平均.

建议回复:

在写代码前,先把实现原理介绍一下,然后再开始写代码:

//创建一个数组
let arr = Array(100000)
  .fill(0)
  .map((item, index) => index + 1);
console.log(arr);

arr.sort((a, b) => (Math.random() > 0.5 ? -1 : 1));
console.log(arr);

Fisher–Yates 经典洗牌算法原理很简单,就是遍历数组元素,然后将当前元素与其后随机位置的元素进行交换,从代码中也可以看出,这样乱序的就会更加彻底.

// ES5 +  Fisher–Yates 经典洗牌算法实现
function shuffle(array) {
  let arrayLength = array.length,
    randomIndex, //随机数
    tempItem; //临时存储元素
  for (let i = arrayLength - 1; i >= 0; i--) {
    randomIndex = Math.floor(Math.random() * (i + 1));
    //i = 0 的时候不需要再做交换了,这样写可以少一次交换
    tempItem = array[randomIndex];
    array[randomIndex] = array[i];
    array[i] = tempItem;
  }
  return array;
}
// ES6 +  Fisher–Yates 经典洗牌算法实现
function shuffle(a) {
  for (let i = a.length; i; i--) {
    //遍历数组元素
    let randomIndex = Math.floor(Math.random() * (i + 1));
    //产生随机数 i = 0 的时候不需要再做交换了,这样写可以少一次交换
    [a[i - 1], a[randomIndex]] = [a[randomIndex], a[i - 1]];
    //将当前元素与其后随机位置的元素进行交换
  }
  return a;
}

reduce 代替 map 和 filter

const _arr = [0, 1, 2];

// map
const arr = _arr.map((v) => v * 2);
const arr = _arr.reduce((t, v) => {
  t.push(v * 2);
  return t;
}, []);
// arr => [0, 2, 4]

// filter
const arr = _arr.filter((v) => v > 0);
const arr = _arr.reduce((t, v) => {
  v > 0 && t.push(v);
  return t;
}, []);
// arr => [1, 2]

// map和filter
const arr = _arr.map((v) => v * 2).filter((v) => v > 2);
const arr = _arr.reduce((t, v) => {
  v = v * 2;
  v > 2 && t.push(v);
  return t;
}, []);
// arr => [4]

function Func() {
  if (a === b) {

      console.log('y');
    };
  }
  return Func();
}
