JavaScript 设置 CSS 样式

CSS
JavaScript
发布日期:; 2015-11-13

用 JavaScript 来动态设置 CSS 样式, 常见的有以下几种

ElementCSSInlineStyle.style 直接设置 style 的属性

某些情况用这个设置 !important 值无效

如果属性有’-‘号, 就写成驼峰的形式 (如 textAlign) 如果想保留 - 号, 就中括号的形式

element.style['text-align'] = '100px';

element.style.height = '100px';

Element.setAttribute() 直接设置属性

element.setAttribute('height', 100);
element.setAttribute('height', '100px');

Element.setAttribute() 设置 style 的属性

element.setAttribute('style', 'height: 100px !important');

CSSStyleDeclaration.setProperty() 设置多个样式

element.setAttribute('style', 'color:red; border: 1px solid blue;');

如果要设置! important, 推荐用这种方法设置第三个参数

element.style.setProperty('height', '300px', 'important');

Element.className 来修改样式表的类名

因 JavaScript 获取不到 CSS 的伪元素, 所以可以通过改变伪元素父级的 class 来动态更改伪元素的样式

element.className = 'blue';
element.className += 'blue fb';

使用 obj.style.cssText 来修改嵌入式的 css

element.style.cssText = 'color: blue; border: 1px solid black';

创建引入新的 CSS 样式文件

function addNewStyle(newStyle) {
  var styleElement = document.getElementById('styles_js');

  if (!styleElement) {
    styleElement = document.createElement('style');
    styleElement.type = 'text/css';
    styleElement.id = 'styles_js';
    document.getElementsByTagName('head')[0].appendChild(styleElement);
  }

  styleElement.appendChild(document.createTextNode(newStyle));
}

addNewStyle('.box {height: 100px !important;}');

使用 CSSStyleSheet.addRule() CSSStyleSheet.insertRule()

// 在原有样式操作
document.styleSheets[0].addRule('.box', 'height: 100px');
document.styleSheets[0].insertRule('.box {height: 100px}', 0);

// 或者插入新样式时操作
var styleEl = document.createElement('style'),
  styleSheet = styleEl.sheet;

styleSheet.addRule('.box', 'height: 100px');
styleSheet.insertRule('.box {height: 100px}', 0);

document.head.appendChild(styleEl);
